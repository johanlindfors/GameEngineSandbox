cmake_minimum_required(VERSION 3.10)

get_property(PLATFORM_ASSETS_DIRECTORY GLOBAL PROPERTY GLOBAL_PLATFORM_ASSETS_DIRECTORY)
get_property(COMMON_ASSETS_DIRECTORY GLOBAL PROPERTY GLOBAL_COMMON_ASSETS_DIRECTORY)
get_property(ASSETS GLOBAL PROPERTY GLOBAL_ASSETS)

set(PLATFORM_ASSETS_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/${PLATFORM}")
set(COMMON_ASSETS_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/common")

file(GLOB_RECURSE PLATFORM_ASSETS
    ${PLATFORM_ASSETS_DIRECTORY}/*
)

list(APPEND ASSETS ${PLATFORM_ASSETS})

set_property(GLOBAL PROPERTY GLOBAL_PLATFORM_ASSETS_DIRECTORY ${PLATFORM_ASSETS_DIRECTORY})
set_property(GLOBAL PROPERTY GLOBAL_COMMON_ASSETS_DIRECTORY ${COMMON_ASSETS_DIRECTORY})
set_property(GLOBAL PROPERTY GLOBAL_ASSETS ${ASSETS})
